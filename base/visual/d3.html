# d3 学习笔记

# path绘制，显示动画效果

  • 使用stroke-dashoffset, stroke-dasharray和path.getTotalLength() (获取路径的长度) 来动画绘制路径
<svg>
  <defs>
    <marker id="markerArrow" markerWidth="10" markerHeight="10" refX="10" refY="6" orient="auto">
      <path d="M2,2 L10,6 L2,10 L6,6 L2,2" :style="{fill: '#ababab'}" />
    </marker>
  </defs>
  <g></g>
</svg>

通过js实现

const path = d3.select('svg')
  .append('path')
  .attr('d', 'M 267 42 L 267 27Q 267 12 282 12 L 948 12 Q 963 12 963 27 L 963 42')
  .attr('stroke', '#ababab')
  .attr('fill', 'transparent')
const totalLength = path.node().getTotalLength()

// 实现动画
path
  // 绘制实线-虚线，长度为总长度 此时还是可以看到整个path
  .attr('stroke-dasharray', totalLength)
  // 添加stroke-dashoffset，偏移为总长度，显示为虚线
  .attr('stroke-dashoffset', totalLength)
  .transition()
  .duration(1200)
  .ease(d3.easeCubic)
  .delay(200)
  // 通过改变stroke-dashoffset，实现绘制的动画效果
  .attr('stroke-dashoffset', 0)
  .on('end', _ => path.attr('marker-end', 'url(#markerArrow)'))

通过css实现

path {
  stroke-dasharray: 1000;
  stroke-dashoffset: 1000;
  animation: dash 1.2s linear infinite;
}

@keyframes dash {
  to {
    stroke-dashoffset: 0;
  }
}
